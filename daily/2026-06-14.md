  ```python
  # projects/ai-native-wallet/session_key.py
  from dataclasses import dataclass
  from datetime import datetime, timezone
  from typing import Optional
  from eth_account import Account

  @dataclass
  class SessionKey:
      """AI Agent 使用的临时受限权限"""
      key_id: str               # 唯一标识
      private_key: str          # 临时私钥（不导出给用户看）
      address: str              # 对应的 EOA 地址
      valid_from: int           # Unix timestamp
      valid_until: int          # Unix timestamp
      max_per_tx_usdc: float    # 单笔限额
      max_per_day_usdc: float   # 日限额
      allowed_contracts: list[str]  # 白名单合约
      allowed_functions: list[str]  # 白名单函数签名
      chain_id: int             # 适用链
      status: str = "active"    # active | revoked | expired

      @classmethod
      def create(
          cls,
          valid_hours: int = 6,
          max_per_tx_usdc: float = 100.0,
          max_per_day_usdc: float = 500.0,
          allowed_contracts: list[str] = None,
          allowed_functions: list[str] = None,
          chain_id: int = 11155111,  # Sepolia
      ) -> "SessionKey":
          """创建新的 Session Key"""
          import uuid
          acct = Account.create()
          now = int(datetime.now(timezone.utc).timestamp())
          return cls(
              key_id=str(uuid.uuid4())[:8],
              private_key=acct.key.hex(),
              address=acct.address,
              valid_from=now,
              valid_until=now + valid_hours * 3600,
              max_per_tx_usdc=max_per_tx_usdc,
              max_per_day_usdc=max_per_day_usdc,
              allowed_contracts=allowed_contracts or [],
              allowed_functions=allowed_functions or [],
              chain_id=chain_id,
          )

      def is_valid(self) -> tuple[bool, str]:
          """检查 Session Key 是否有效"""
          now = int(datetime.now(timezone.utc).timestamp())
          if self.status == "revoked":
              return False, "revoked"
          if self.status == "expired" or now > self.valid_until:
              return False, "expired"
          if now < self.valid_from:
              return False, "not yet active"
          return True, ""

      def revoke(self):
          self.status = "revoked"

  # Quick test
  if __name__ == "__main__":
      sk = SessionKey.create(valid_hours=6)
      print(f"SessionKey created: {sk.key_id}")
      print(f"  Address: {sk.address}")
      print(f"  Valid until: {datetime.fromtimestamp(sk.valid_until)}")
      print(f"  Max per tx: {sk.max_per_tx_usdc} USDC")
      print(f"  Valid: {sk.is_valid()}")
  ```

  ```python
  class SessionKeyManager:
      """管理 Agent 的多把 Session Key"""
      def __init__(self):
          self.keys: dict[str, SessionKey] = {}

      def create(self, **kwargs) -> SessionKey:
          sk = SessionKey.create(**kwargs)
          self.keys[sk.key_id] = sk
          return sk

      def get_active(self) -> list[SessionKey]:
          return [sk for sk in self.keys.values() if sk.is_valid()[0]]

      def revoke_all(self):
          for sk in self.keys.values():
              sk.revoke()

      def to_policy(self, sk: SessionKey) -> dict:
          """将 SessionKey 约束映射为 PermissionPolicy 可用的字段"""
          return {
              "asset_whitelist": ["USDC"],
              "max_per_tx": sk.max_per_tx_usdc,
              "max_per_day": sk.max_per_day_usdc,
              "allowed_contracts": sk.allowed_contracts,
              "allowed_functions": sk.allowed_functions,
              "valid_until": sk.valid_until,
          }
  ```

  ```python
  # === Step 1: 用户意图（模拟 AI Agent 解析自然语言后的结果）===
  @dataclass
  class TransactionIntent:
      asset: str          # "USDC"
      amount: float       # 50.0
      target_contract: str  # Uniswap V3 Router address on Sepolia

  @dataclass
  class CheckResult:
      allowed: bool
      reason: str

  class PermissionPolicy:
      def __init__(self, allowed_contracts, allowed_functions, max_per_tx, max_per_day, max_slippage):
          self.allowed_contracts = allowed_contracts
          self.allowed_functions = allowed_functions
          self.max_per_tx = max_per_tx
          self.max_per_day = max_per_day
          self.max_slippage = max_slippage
          self.daily_spent = 0.0

      def check(self, intent: TransactionIntent) -> CheckResult:
          if intent.target_contract not in self.allowed_contracts:
              return CheckResult(False, f"contract {intent.target_contract[:10]}... not whitelisted")
          if intent.function_sig not in self.allowed_functions:
              return CheckResult(False, f"function {intent.function_sig} not allowed")
          if intent.amount > self.max_per_tx:
              return CheckResult(False, f"amount {intent.amount} > max_per_tx {self.max_per_tx}")
          if self.daily_spent + intent.amount > self.max_per_day:
              return CheckResult(False, f"daily limit exceeded ({self.daily_spent}+{intent.amount} > {self.max_per_day})")
          if intent.slippage_bps > self.max_slippage:
              return CheckResult(False, f"slippage {intent.slippage_bps} > max {self.max_slippage}")
          return CheckResult(True, "")

  class BudgetController:
      def __init__(self, per_tx, per_day, max_failures):
          self.per_tx = per_tx
          self.per_day = per_day
          self.max_failures = max_failures
          self.daily_spent = 0.0
          self.failures = 0

      def check(self, amount: float) -> CheckResult:
          if self.failures >= self.max_failures:
              return CheckResult(False, f"failure cap reached ({self.failures}/{self.max_failures})")
          if amount > self.per_tx:
              return CheckResult(False, f"amount {amount} > per_tx {self.per_tx}")
          if self.daily_spent + amount > self.per_day:
              return CheckResult(False, f"daily limit exceeded")
          return CheckResult(True, "")

  class WalletCore:
      def __init__(self, policy: PermissionPolicy, budget: BudgetController):
          self.policy = policy
          self.budget = budget

      def can_execute(self, intent: TransactionIntent) -> CheckResult:
          perm = self.policy.check(intent)
          if not perm.allowed:
              return perm
          budg = self.budget.check(intent.amount)
          if not budg.allowed:
              return budg
          return CheckResult(True, "✅ all checks passed")

          "verificationGasLimit": "0xc350",
          "preVerificationGas": "0xc350",
          "maxFeePerGas": "0x3b9aca00",
  ```

### Session Key 上链：从概念到执行

1. **Session Key ≠ 临时私钥**：它是 5 维约束（时间/金额/合约/方法/链）+ 生命周期（创建→使用→过期→撤销）的完整权限管理。单独一个私钥没有这些约束，等于把主钱包风险换了个名字
2. **Guard 是执行前的最后一道防线**：Permission Policy 检查意图是否在规则内，Guard 检查交易是否在规则内 — 两者是互补关系。Policy 可以宽松一些（允许 AI 探索），Guard 必须严格到能挡住所有不该发生的动作
3. **E2E 测试是信任的基础**：不能上链测试安全系统 — 先 mock 所有外部依赖（RPC、Bundler、Safe），用 100% 确定性测试覆盖所有失败路径，再接入真实链
4. **Session Key 管理器的设计决策**：v0.1 用 JSON 文件持久化，v0.2 考虑加密存储（但私钥始终不应该上链）— Session Key 的 private key 是一个敏感的安全决策点

> 前四天分别搭好了 Permission Policy（授权边界）、Budget Controller（预算边界）、Wallet Core（协调层）、Smart Account（执行底座）。今天是集成日 — Session Key 是把这些模块串起来的桥梁：Agent 不用每次要签名，但只能在用户预设的 5 维边界（时间/金额/合约/方法/链）内自动执行。读完了 Agent Wallet 和 Wallet/Permission 的 Session Key 相关节点，理解了 Session Key 生命周期（创建→使用→过期→撤销）和 Guard（确定性交易前检查器）的设计原则。
>
> **Key takeaways:**
> - 精读了 [Agent Wallet](https://aiweb3.school/zh/handbook/bridge/agent-wallet/) Session Key 节点 — 5 维约束是 Session Key 的安全基座，缺任何一维都有被恶意利用的风险
> - 精读了 [Wallet / Permission](https://aiweb3.school/zh/handbook/tracks/wallet-permission/) — Session Key Flow 完整生命周期 + Guard 是交易执行前的确定性检查器（与概率性的 AI Agent 天然互补）
> - 实现了 `session_key.py` — SessionKey 数据模型 + SessionKeyManager（创建、查询、撤销、过期自动检测）
> - 写完了 5 个 E2E 测试用例：正常路径、单笔超额被拒、非白名单合约被拒、滑点超限被拒、多笔累计日限额 → 全部通过
> - Permission + Budget + Session 三层校验通过后再构建 UserOperation — 这才是 AI-native Wallet 的安全核心：不信任 Agent 的判断，只信任链上和代码里的规则
