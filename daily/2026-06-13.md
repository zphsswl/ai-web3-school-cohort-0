- [ ] **实现 `smart_account.py` — UserOperation 构建器** (15 min)
  ```python
  # projects/ai-native-wallet/smart_account.py
  from web3 import Web3
  from dataclasses import dataclass, field
  from typing import Optional
  import json

  # 连接 Sepolia
  SEPOLIA_RPC = "https://rpc.sepolia.org"
  ENTRY_POINT_ADDRESS = "0x5FF137D4b0FDCD49DcA30c7CF57E578a026d2789"

  @dataclass
  class UserOperation:
      """ERC-4337 UserOperation 结构"""
      sender: str                          # Smart Account 地址
      nonce: int                           # 从 EntryPoint 获取
      initCode: bytes = b''                # 首次部署时使用
      callData: bytes                      # 目标调用数据
      callGasLimit: int = 200000
      verificationGasLimit: int = 50000
      preVerificationGas: int = 50000
      maxFeePerGas: int = 0
      maxPriorityFeePerGas: int = 0
      paymasterAndData: bytes = b''
      signature: bytes = b''

      def to_dict(self) -> dict:
          """转换为 bundler 接受的 JSON 格式"""
          return {
              "sender": self.sender,
              "nonce": hex(self.nonce),
              "initCode": "0x" + self.initCode.hex() if self.initCode else "0x",
              "callData": "0x" + self.callData.hex(),
              "callGasLimit": hex(self.callGasLimit),
              "verificationGasLimit": hex(self.verificationGasLimit),
              "preVerificationGas": hex(self.preVerificationGas),
              "maxFeePerGas": hex(self.maxFeePerGas),
              "maxPriorityFeePerGas": hex(self.maxPriorityFeePerGas),
              "paymasterAndData": "0x" + self.paymasterAndData.hex() if self.paymasterAndData else "0x",
              "signature": "0x" + self.signature.hex() if self.signature else "0x",
          }


  class UserOpBuilder:
      """从 TransactionIntent 构建 UserOperation"""

      def __init__(self, w3: Web3, entry_point: str = ENTRY_POINT_ADDRESS):
          self.w3 = w3
          self.entry_point = Web3.to_checksum_address(entry_point)

      def build(
          self,
          sender: str,
          to_contract: str,
          calldata: bytes,
          value: int = 0,
          paymaster: str = "",
      ) -> UserOperation:
          """从意图构建一个 UserOperation"""
          # 获取 nonce（简化版：从 EntryPoint 合约查询）
          nonce = self._get_nonce(sender)

          # 获取 gas 价格
          fee_data = self.w3.eth.fee_history(1, 'latest')

          return UserOperation(
              sender=Web3.to_checksum_address(sender),
              nonce=nonce,
              callData=self._encode_exec_call(to_contract, value, calldata),
              maxFeePerGas=self.w3.eth.gas_price,  # 简化
              maxPriorityFeePerGas=2_000_000_000,   # 2 gwei
          )

      def _get_nonce(self, sender: str) -> int:
          """从 EntryPoint 合约获取 sender 的 nonce"""
          abi = '[{"inputs":[{"name":"sender","type":"address"}],"name":"getNonce","outputs":[{"name":"nonce","type":"uint256"}],"stateMutability":"view","type":"function"}]'
          contract = self.w3.eth.contract(address=self.entry_point, abi=json.loads(abi))
          return contract.functions.getNonce(sender).call()

      def _encode_exec_call(self, to: str, value: int, calldata: bytes) -> bytes:
          """编码 Safe 的 execTransaction 调用（简化）"""
          # Safe execTransaction(from, to, value, data, operation, ...)
          # 简化版：直接编码目标调用
          return calldata

      def estimate_gas(self, user_op: UserOperation) -> tuple[int, int]:
          """估算 callGasLimit 和 verificationGasLimit"""
          # 简化：返回默认值
          return user_op.callGasLimit, user_op.verificationGasLimit


  # === Quick Test ===
  if __name__ == "__main__":
      w3 = Web3(Web3.HTTPProvider(SEPOLIA_RPC))
      print(f"Connected to Sepolia: {w3.is_connected()}")
      print(f"EntryPoint: {ENTRY_POINT_ADDRESS}")

      # 替换为你的 Safe 地址
      SAFE_ADDRESS = "0xYOUR_SAFE_ADDRESS"

      builder = UserOpBuilder(w3)
      # 构造一个简单的 UserOperation（调用自己 — 空 calldata）
      user_op = builder.build(
          sender=SAFE_ADDRESS,
          to_contract=SAFE_ADDRESS,
          calldata=b'',
          value=0,
      )
      print(f"UserOperation: {json.dumps(user_op.to_dict(), indent=2)}")
  ```

## 📝 Learning Notes

<!-- Your insights from today's reading and building -->

### Account Abstraction × AI-native Wallet 映射

1. **EOA → Smart Account 是质的飞跃**: 不是 "换个钱包界面"，而是账户从单密钥变成可编程规则引擎 — 这正是 AI-native Wallet 需要的基础设施
2. **Session Key 不等于 "临时私钥"**: 它是一组受限权限的载体 — 时间、金额、合约、方法、链，5 个维度缺一个是安全隐患
3. **EntryPoint + Bundler 是基础设施依赖**: 你的产品依赖 bundler 的稳定性和正确性 — 选一个可靠 bundler（Pimlico/Stackup）是关键架构决策
4. **Paymaster 不是免费午餐**: 赞助 gas 需要风控 — 哪些方法可赞助、每用户额度、防 spam、失败成本

### 实现决策

- **Safe{Wallet} 选型**: 选择 Safe 而非 ZeroDev/Biconomy，因为 Safe 是行业标准 + 模块化架构 + 多签原生支持
- **Sepolia 先验证，后续跨链**: Sepolia 成本极低，可充分测试所有路径后再上主网
- **v0.1 先做 1/1 Safe**: 单人控制，v0.2 再加 Session Key 模块，v0.3 加多签治理
- **Bundler**: 先用 Pimlico 免费 Tier（Sepolia 每天 100 次免费 UserOperation），后续按需升级
